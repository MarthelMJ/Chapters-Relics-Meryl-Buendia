// DISPLAY
function displayProducts(list){
    let grid = document.getElementById("productGrid");
    grid.innerHTML="";
    list.forEach((p,i)=>{
        grid.innerHTML+=`
        <div class="card" onclick="openProduct(${i})">
            <img src="${p.images[0]}">
            <h3>${p.name}</h3>
            <p class="price">₱${p.price}</p>
            <p class="desc">${p.desc}</p>
            <p class="rating">⭐ ${p.rating} | Sold ${p.sold}</p>
        </div>`;
    });
}
function addFavorite(i){
    if(!favorites.includes(products[i])){
        favorites.push(products[i]);
        alert("Added to favorites ❤️");
    }
}

function toggleFavorites(){
    displayProducts(favorites);
}

function addToCart(name,price){
    cart.push({name,price});
    updateCart();
}

function updateCart(){
    let container = document.getElementById("cartItems");
    container.innerHTML="";

    cart.forEach(item=>{
        container.innerHTML+=`
        <p>${item.name} - ₱${item.price}</p>
        `;
    });
}

function toggleCart(){
    document.getElementById("cartPanel").classList.toggle("show");
}

// OPEN PRODUCT
function openProduct(i){
    currentProduct = products[i];
    document.getElementById("mainShop").style.display="none";
    document.getElementById("productPage").style.display="block";

    document.getElementById("pName").innerText=currentProduct.name;
    document.getElementById("pPrice").innerText="₱"+currentProduct.price;
    document.getElementById("pDesc").innerText=currentProduct.desc;

    document.getElementById("mainImage").src=currentProduct.images[0];

    let thumbs=document.getElementById("thumbs");
    thumbs.innerHTML="";
    currentProduct.images.forEach(img=>{
        thumbs.innerHTML+=`<img src="${img}" onclick="changeImage('${img}')">`;
    });
}

// IMAGE CHANGE
function changeImage(src){
    document.getElementById("mainImage").src=src;
}

// NAVIGATION
function goBack(){
    document.getElementById("productPage").style.display="none";
    document.getElementById("mainShop").style.display="block";
}

function goToShop(){
    document.getElementById("landingPage").style.display="none";
    document.getElementById("mainShop").style.display="block";
}

// FILTER
function filterCategory(cat){
    if(cat=="all") displayProducts(products);
    else displayProducts(products.filter(p=>p.category==cat));
}
